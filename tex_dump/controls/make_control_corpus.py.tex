"""
Control Corpus Generator - COMPLETE VERSION

Creates deterministic control corpora with manifest to ensure:
1. All seeds see EXACT same articles (no input variance)
2. Article lengths match real corpus distribution
3. Proper separation of control types

Three control types:
- CONSTANT: Identical article (architectural baseline, should → 0% residual)
- SHUFFLED: Random word order (noise baseline, expect ~20-30% residual)
- RANDOM: Random tokens (hard baseline, expect ~10-20% residual)

Critical: Generates MANIFEST so different seeds process identical articles.
"""

import numpy as np
import json
from pathlib import Path
from typing import List, Dict, Tuple
import argparse
import hashlib


def compute_manifest_hash(articles: List[Dict]) -> str:
    """Compute deterministic hash of article manifest."""
    # Hash based on article order and content
    content_str = "".join([a['content'] + a['url'] for a in articles])
    return hashlib.md5(content_str.encode()).hexdigest()[:8]


def make_constant_corpus(n_articles: int = 500, seed: int = 42) -> Tuple[List[Dict], str]:
    """
    CONSTANT: Identical article repeated N times.
    
    Expected: 0% residual variance (perfect observer agreement)
    Tests: Architectural baseline (can system detect zero variance?)
    """
    np.random.seed(seed)
    
    base_text = """
    The ongoing conflict in the Gaza Strip continues to evolve as international 
    observers monitor the humanitarian situation. Multiple stakeholders have 
    expressed concerns about civilian casualties and infrastructure damage. 
    Diplomatic efforts remain ongoing as regional tensions persist. The 
    international community has called for immediate ceasefire negotiations 
    while humanitarian aid organizations work to address urgent needs on the 
    ground. Regional powers continue diplomatic engagement as the situation 
    develops with significant implications for stability.
    """
    
    articles = []
    for i in range(n_articles):
        articles.append({
            'content': base_text.strip(),
            'title': f'Control Constant Article {i}',
            'source': 'ControlConstant',
            'published_at': f'2024-01-01T{i%24:02d}:00:00Z',
            'url': f'https://control.constant/{i}',
            'control_class': 'constant',
            'control_index': i
        })
    
    manifest_hash = compute_manifest_hash(articles)
    return articles, manifest_hash


def make_shuffled_corpus(n_articles: int = 500, seed: int = 42) -> Tuple[List[Dict], str]:
    """
    SHUFFLED: Same vocabulary, random word order per article.
    
    Expected: 20-30% residual variance (baseline noise from broken syntax)
    Tests: Whether real variance exceeds simple word-order randomness
    """
    np.random.seed(seed)
    
    base_text = """
    The ongoing conflict in the Gaza Strip continues to evolve as international 
    observers monitor the humanitarian situation. Multiple stakeholders have 
    expressed concerns about civilian casualties and infrastructure damage. 
    Diplomatic efforts remain ongoing as regional tensions persist. The 
    international community has called for immediate ceasefire negotiations 
    while humanitarian aid organizations work to address urgent needs on the 
    ground. Regional powers continue diplomatic engagement as the situation 
    develops with significant implications for stability.
    """
    
    tokens = base_text.strip().split()
    
    articles = []
    for i in range(n_articles):
        # IMPORTANT: Use article-specific seed for deterministic shuffle
        article_seed = seed + i
        rng = np.random.RandomState(article_seed)
        shuffled_tokens = rng.permutation(tokens)
        
        articles.append({
            'content': ' '.join(shuffled_tokens),
            'title': f'Control Shuffled Article {i}',
            'source': 'ControlShuffled',
            'published_at': f'2024-01-01T{i%24:02d}:00:00Z',
            'url': f'https://control.shuffled/{i}',
            'control_class': 'shuffled',
            'control_index': i,
            'shuffle_seed': article_seed  # For reproducibility
        })
    
    manifest_hash = compute_manifest_hash(articles)
    return articles, manifest_hash


def make_random_corpus(n_articles: int = 500, seed: int = 42) -> Tuple[List[Dict], str]:
    """
    RANDOM: Truly random token sequences (NO domain overlap).
    
    Expected: 10-20% residual variance (hard baseline, pure noise)
    Tests: Lower bound on observer variance
    
    CRITICAL FIX (Dec 2024): Removed domain-specific vocabulary (Gaza, Israel, etc.)
    to prevent contamination. Now uses ONLY generic English words.
    This ensures Real ≠ Random by removing domain overlap.
    """
    np.random.seed(seed)
    
    # FIXED: Generic vocabulary ONLY (NO conflict/Gaza/Israel words!)
    vocab = [
        # Common function words
        'the', 'and', 'is', 'to', 'in', 'of', 'a', 'that', 'it', 'with',
        'as', 'for', 'on', 'was', 'are', 'by', 'this', 'from', 'or', 'at',
        'be', 'have', 'not', 'but', 'can', 'will', 'would', 'could', 'should',
        'there', 'their', 'what', 'so', 'up', 'out', 'if', 'about', 'who',
        
        # Generic nouns (NO political/conflict terms)
        'time', 'person', 'year', 'way', 'day', 'thing', 'man', 'world', 'life',
        'hand', 'part', 'child', 'eye', 'woman', 'place', 'work', 'week', 'case',
        'point', 'government', 'company', 'number', 'group', 'problem', 'fact',
        'water', 'room', 'mother', 'area', 'money', 'story', 'result', 'door',
        
        # Generic verbs
        'make', 'get', 'give', 'tell', 'become', 'leave', 'feel', 'put', 'bring',
        'begin', 'keep', 'hold', 'write', 'stand', 'hear', 'let', 'mean', 'set',
        'meet', 'run', 'move', 'like', 'live', 'believe', 'happen', 'must', 'start',
        
        # Generic adjectives
        'good', 'new', 'first', 'last', 'long', 'great', 'little', 'own', 'other',
        'old', 'right', 'big', 'high', 'different', 'small', 'large', 'next', 'early',
        'young', 'important', 'few', 'public', 'bad', 'same', 'able', 'local'
    ]
    
    # Match base text length (~70 tokens)
    n_tokens = 70
    
    articles = []
    for i in range(n_articles):
        # IMPORTANT: Use article-specific seed for deterministic generation
        article_seed = seed + i
        rng = np.random.RandomState(article_seed)
        random_tokens = rng.choice(vocab, size=n_tokens, replace=True)
        
        articles.append({
            'content': ' '.join(random_tokens),
            'title': f'Control Random Article {i}',
            'source': 'ControlRandom',
            'published_at': f'2024-01-01T{i%24:02d}:00:00Z',
            'url': f'https://control.random/{i}',
            'control_class': 'random',
            'control_index': i,
            'generation_seed': article_seed  # For reproducibility
        })
    
    manifest_hash = compute_manifest_hash(articles)
    return articles, manifest_hash


def save_corpus_with_manifest(articles: List[Dict], manifest_hash: str, 
                               output_path: Path, control_type: str):
    """Save corpus and manifest for reproducibility."""
    output_path.parent.mkdir(parents=True, exist_ok=True)
    
    # Save articles
    with open(output_path, 'w', encoding='utf-8') as f:
        for article in articles:
            f.write(json.dumps(article) + '\n')
    
    # Save manifest
    manifest = {
        'control_type': control_type,
        'n_articles': len(articles),
        'manifest_hash': manifest_hash,
        'article_hashes': [hashlib.md5(a['content'].encode()).hexdigest()[:8] 
                          for a in articles]
    }
    
    manifest_path = output_path.parent / f'{control_type}_manifest.json'
    with open(manifest_path, 'w') as f:
        json.dump(manifest, f, indent=2)
    
    print(f"  ✓ Saved {len(articles)} articles: {output_path}")
    print(f"  ✓ Manifest hash: {manifest_hash}")
    print(f"  ✓ Manifest file: {manifest_path}")


def verify_manifest(corpus_path: Path, manifest_path: Path) -> bool:
    """Verify corpus matches its manifest."""
    # Load corpus
    articles = []
    with open(corpus_path, 'r', encoding='utf-8') as f:
        for line in f:
            articles.append(json.loads(line))
    
    # Load manifest
    with open(manifest_path, 'r') as f:
        manifest = json.load(f)
    
    # Verify count
    if len(articles) != manifest['n_articles']:
        print(f"  ✗ Article count mismatch: {len(articles)} vs {manifest['n_articles']}")
        return False
    
    # Verify hashes
    for i, article in enumerate(articles):
        expected_hash = manifest['article_hashes'][i]
        actual_hash = hashlib.md5(article['content'].encode()).hexdigest()[:8]
        if expected_hash != actual_hash:
            print(f"  ✗ Article {i} hash mismatch")
            return False
    
    print(f"  ✓ Manifest verification passed")
    return True


def print_samples(controls: Dict[str, List[Dict]]):
    """Print sample from each control class."""
    print(f"\n{'='*70}")
    print("CONTROL SAMPLES (First article from each)")
    print(f"{'='*70}")
    
    for cls in ['constant', 'shuffled', 'random']:
        if cls not in controls:
            continue
            
        article = controls[cls][0]
        print(f"\n{cls.upper()}:")
        print(f"  Text: {article['content'][:100]}...")
        print(f"  Length: {len(article['content'].split())} tokens")
        print(f"  Source: {article['source']}")
        print(f"  Index: {article['control_index']}")


def main():
    parser = argparse.ArgumentParser(
        description='Generate deterministic control corpora with manifests'
    )
    parser.add_argument('--n-articles', type=int, default=500,
                       help='Articles per control class (default: 500)')
    parser.add_argument('--seed', type=int, default=42,
                       help='Master random seed (default: 42)')
    parser.add_argument('--output-dir', type=str, default='data',
                       help='Output directory (default: data)')
    parser.add_argument('--verify', action='store_true',
                       help='Verify manifests after generation')
    
    args = parser.parse_args()
    
    output_dir = Path(args.output_dir)
    
    print(f"{'='*70}")
    print("CONTROL CORPUS GENERATOR - DETERMINISTIC MODE")
    print(f"{'='*70}")
    print(f"Articles per class: {args.n_articles}")
    print(f"Master seed: {args.seed}")
    print(f"Output directory: {output_dir.absolute()}")
    
    # Generate all controls
    print(f"\n{'='*70}")
    print("GENERATING CONTROL CORPORA")
    print(f"{'='*70}")
    
    controls = {}
    manifests = {}
    
    print("\n[1/3] CONSTANT corpus...")
    controls['constant'], manifests['constant'] = make_constant_corpus(
        args.n_articles, args.seed
    )
    
    print("\n[2/3] SHUFFLED corpus...")
    controls['shuffled'], manifests['shuffled'] = make_shuffled_corpus(
        args.n_articles, args.seed
    )
    
    print("\n[3/3] RANDOM corpus...")
    controls['random'], manifests['random'] = make_random_corpus(
        args.n_articles, args.seed
    )
    
    print(f"\n✓ Generated all control classes")
    
    # Save
    print(f"\n{'='*70}")
    print("SAVING CONTROL CORPORA")
    print(f"{'='*70}")
    
    for control_type, articles in controls.items():
        output_path = output_dir / f'control_{control_type}.jsonl'
        save_corpus_with_manifest(
            articles, 
            manifests[control_type],
            output_path,
            control_type
        )
    
    # Print samples
    print_samples(controls)
    
    # Verify if requested
    if args.verify:
        print(f"\n{'='*70}")
        print("VERIFYING MANIFESTS")
        print(f"{'='*70}")
        
        all_valid = True
        for control_type in controls.keys():
            corpus_path = output_dir / f'control_{control_type}.jsonl'
            manifest_path = output_dir / f'{control_type}_manifest.json'
            
            print(f"\nVerifying {control_type}...")
            valid = verify_manifest(corpus_path, manifest_path)
            all_valid = all_valid and valid
        
        if all_valid:
            print(f"\n✓✓✓ All manifests verified!")
        else:
            print(f"\n✗ Some manifests failed verification")
            return 1
    
    # Summary
    print(f"\n{'='*70}")
    print("CONTROL CORPUS GENERATION COMPLETE")
    print(f"{'='*70}")
    print(f"\nGenerated:")
    for control_type, articles in controls.items():
        print(f"  {control_type:10s}: {len(articles):4d} articles (hash: {manifests[control_type]})")
    
    print(f"\nOutput directory: {output_dir.absolute()}")
    
    print(f"\n{'='*70}")
    print("NEXT STEPS")
    print(f"{'='*70}")
    print(f"\n1. Run experiments on controls (INTRA-GRU mode):")
    print(f"   python run_experiments.py --corpus control_constant --mode enhanced --gru-mode intra --seeds 42 43 44 45 46")
    print(f"   python run_experiments.py --corpus control_shuffled --mode enhanced --gru-mode intra --seeds 42 43 44 45 46")
    print(f"   python run_experiments.py --corpus control_random --mode enhanced --gru-mode intra --seeds 42 43 44 45 46")
    
    print(f"\n2. Run experiments on controls (INTER-GRU mode):")
    print(f"   python run_experiments.py --corpus control_constant --mode enhanced --gru-mode inter --seeds 42 43 44 45 46")
    print(f"   python run_experiments.py --corpus control_shuffled --mode enhanced --gru-mode inter --seeds 42 43 44 45 46")
    print(f"   python run_experiments.py --corpus control_random --mode enhanced --gru-mode inter --seeds 42 43 44 45 46")
    
    print(f"\n3. Compare results:")
    print(f"   python compare_controls.py --gru-mode intra")
    print(f"   python compare_controls.py --gru-mode inter")
    
    print(f"\n{'='*70}")
    
    return 0


if __name__ == "__main__":
    exit(main())
