"""
Procrustes Utilities for Observer Comparison - CORRECTED

This module provides Procrustes functions for the belief transformer pipeline.
Used by analysis scripts like procrustes_alignment.py.

CRITICAL FIXES (Jan 2026):
--------------------------
1. Energy-based variance decomposition (no double-counting)
2. Pure orthogonal Procrustes (no fake scale factor)
3. Separate HIGH-D (512D) and 2D (visualization) functions

Usage:
    from core.procrustes import procrustes_align, compute_consensus_and_residuals

Reference: ChatGPT analysis (Jan 7, 2026)
"""

import torch
import numpy as np
from scipy.linalg import orthogonal_procrustes
from typing import List, Dict, Tuple
from collections import defaultdict


# ============================================================================
# HIGH-DIMENSIONAL PROCRUSTES (512D) - For semantic analysis
# ============================================================================

def procrustes_align(
    embeddings_list: List[torch.Tensor],
    reference_idx: int = 0
) -> Dict:
    """
    Align multiple observer geometries using orthogonal Procrustes (HIGH-D).
    
    Finds optimal rotation to align each observer to reference,
    minimizing Frobenius norm: ||X_i @ R_i - X_ref||_F
    
    Parameters
    ----------
    embeddings_list : List[Tensor]
        List of [N, D] embedding matrices from different observers (typically D=512)
    reference_idx : int
        Which observer to use as reference (default: first)
    
    Returns
    -------
    dict
        - aligned: List of aligned embeddings [N, D]
        - transformations: List of rotation matrices [D, D]
        - distances_before: Frobenius distances before alignment
        - distances_after: Frobenius distances after alignment
        - reference_idx: Index of reference observer
        - n_observers: Number of observers
    """
    
    n_observers = len(embeddings_list)
    reference = embeddings_list[reference_idx]
    
    # Convert to numpy for scipy
    if torch.is_tensor(reference):
        reference_np = reference.cpu().numpy()
    else:
        reference_np = np.array(reference)
    
    aligned = []
    transformations = []
    distances_before = []
    distances_after = []
    
    for i, emb in enumerate(embeddings_list):
        # Convert to numpy
        if torch.is_tensor(emb):
            emb_np = emb.cpu().numpy()
        else:
            emb_np = np.array(emb)
        
        if i == reference_idx:
            # Reference doesn't need alignment
            aligned.append(torch.FloatTensor(emb_np))
            transformations.append(torch.eye(emb_np.shape[1]))
            distances_before.append(0.0)
            distances_after.append(0.0)
            continue
        
        # Measure Frobenius distance before alignment
        dist_before = np.linalg.norm(emb_np - reference_np, ord='fro')
        distances_before.append(float(dist_before))
        
        # Solve for optimal rotation: min ||emb @ R - reference||_F
        # NOTE: We IGNORE the 'scale' returned by scipy (it's a fit statistic, not a scale factor)
        R, _ = orthogonal_procrustes(emb_np, reference_np)
        
        # Apply ONLY rotation (pure orthogonal Procrustes - no scaling)
        aligned_emb = emb_np @ R
        aligned.append(torch.FloatTensor(aligned_emb))
        transformations.append(torch.FloatTensor(R))
        
        # Measure Frobenius distance after alignment
        dist_after = np.linalg.norm(aligned_emb - reference_np, ord='fro')
        distances_after.append(float(dist_after))
    
    return {
        'aligned': aligned,
        'transformations': transformations,
        'distances_before': distances_before,
        'distances_after': distances_after,
        'reference_idx': reference_idx,
        'n_observers': n_observers
    }


def compute_consensus_and_residuals(
    aligned_embeddings: List[torch.Tensor]
) -> Tuple[torch.Tensor, List[torch.Tensor], Dict]:
    """
    Decompose aligned embeddings into consensus + residuals with CORRECT variance.
    
    CRITICAL FIX: Uses energy-based decomposition (Frobenius norms) to avoid
    double-counting variance. Previous version used .var() which was mathematically wrong.
    
    Energy decomposition:
        E_total = (1/O) Σ ||X_o||²_F
        E_consensus = ||X̄||²_F  
        E_residual = (1/O) Σ ||X_o - X̄||²_F
    
    Then: consensus_fraction = E_consensus / (E_consensus + E_residual)
    
    This guarantees consensus_fraction + residual_fraction = 1.0
    
    Parameters
    ----------
    aligned_embeddings : List[Tensor]
        Aligned embeddings from procrustes_align, each [N, D]
    
    Returns
    -------
    consensus : Tensor [N, D]
        Mean embedding across observers (shared structure)
    residuals : List[Tensor]
        Per-observer residuals [N, D] (observer-specific deviations)
    variance_decomposition : dict
        Energy-based variance statistics (sums to 1.0)
    """
    
    # Stack all observers: [n_observers, N, D]
    stacked = torch.stack(aligned_embeddings, dim=0)
    n_observers, n_articles, dim = stacked.shape
    
    # Consensus = mean across observers
    consensus = stacked.mean(dim=0)  # [N, D]
    
    # Residuals = deviation from consensus
    residuals = []
    for i in range(n_observers):
        residual = stacked[i] - consensus
        residuals.append(residual)
    
    # ========================================================================
    # CORRECTED: Energy-based variance decomposition
    # ========================================================================
    
    # Total energy = mean of per-observer squared Frobenius norms
    observer_energies = [(emb ** 2).sum().item() for emb in aligned_embeddings]
    E_total = float(np.mean(observer_energies))
    
    # Consensus energy = squared Frobenius norm of consensus
    E_consensus = float((consensus ** 2).sum().item())
    
    # Residual energy = mean of per-observer residual squared norms
    residual_energies = [(res ** 2).sum().item() for res in residuals]
    E_residual = float(np.mean(residual_energies))
    
    # Fractions (these sum to 1.0 by construction!)
    total_energy = E_consensus + E_residual
    if total_energy > 1e-10:
        consensus_fraction = E_consensus / total_energy
        residual_fraction = E_residual / total_energy
    else:
        consensus_fraction = 0.0
        residual_fraction = 0.0
    
    # Per-observer breakdown
    per_observer_residual_energy = [float(e) for e in residual_energies]
    per_observer_total_energy = [float(e) for e in observer_energies]
    
    variance_decomposition = {
        # Energy decomposition (Frobenius norms)
        'E_total': E_total,
        'E_consensus': E_consensus,
        'E_residual': E_residual,
        
        # Fractions (sum to 1.0)
        'consensus_fraction': float(consensus_fraction),
        'residual_fraction': float(residual_fraction),
        
        # Per-observer breakdown
        'per_observer_residual_energy': per_observer_residual_energy,
        'per_observer_total_energy': per_observer_total_energy,
        
        # Sanity check (should be ~1.0)
        'fraction_sum': float(consensus_fraction + residual_fraction),
    }
    
    return consensus, residuals, variance_decomposition

# 2D UMAP ALIGNMENT - For visualization only

def procrustes_align_umap(
    coords1: np.ndarray,
    coords2: np.ndarray
) -> Dict:
    """
    Align two 2D coordinate spaces using pure orthogonal Procrustes.
    
    CORRECTED: No longer misuses scipy's 'scale' return value.
    Applies ONLY rotation + translation (standard orthogonal Procrustes).
    
    **WARNING:** This is for VISUALIZATION ONLY!
    Use procrustes_align() for semantic analysis (operates in 512D).
    
    Parameters
    ----------
    coords1 : np.ndarray
        UMAP coordinates from observer 1, shape (N, 2)
    coords2 : np.ndarray
        UMAP coordinates from observer 2, shape (N, 2)
    
    Returns
    -------
    dict
        - aligned_coords2: coords2 after alignment (rotation + translation)
        - rotation_matrix: optimal rotation R (2×2 orthogonal matrix)
        - translation: optimal translation vector (2,)
        - per_point_residuals: Euclidean distances after alignment (N,)
        - total_residual: sum of squared residuals
    """
    assert coords1.shape == coords2.shape, "Coordinate shapes must match"
    assert coords1.shape[1] == 2, "This function is for 2D visualization only"
    
    # Center both coordinate sets
    centroid1 = coords1.mean(axis=0)
    centroid2 = coords2.mean(axis=0)
    
    centered1 = coords1 - centroid1
    centered2 = coords2 - centroid2
    
    # Compute optimal rotation using orthogonal Procrustes
    # Solves: min ||centered1 - centered2 @ R||²_F
    # NOTE: We IGNORE scipy's 'scale' return value (it's not an isotropic scale factor)
    R, _ = orthogonal_procrustes(centered2, centered1)
    
    # Apply transformation: rotate + translate (NO SCALING)
    aligned_coords2 = (centered2 @ R) + centroid1
    
    # Compute residuals (per-point Euclidean distances)
    per_point_residuals = np.linalg.norm(coords1 - aligned_coords2, axis=1)
    total_residual = np.sum(per_point_residuals ** 2)
    
    return {
        'aligned_coords2': aligned_coords2,
        'rotation_matrix': R,
        'translation': centroid1 - centroid2,
        'per_point_residuals': per_point_residuals,
        'total_residual': total_residual
    }

def compare_attention_matrices(
    attn1: np.ndarray,
    attn2: np.ndarray
) -> Dict:
    """
    Compare two attention matrices using element-wise difference.
    
    NOTE: This does NO Procrustes - just element-wise difference + statistics.
    
    Parameters
    ----------
    attn1 : np.ndarray
        Attention matrix from observer 1, shape (N, N)
    attn2 : np.ndarray
        Attention matrix from observer 2, shape (N, N)
    
    Returns
    -------
    dict
        - difference_matrix: attn1 - attn2
        - per_row_residuals: row-wise L2 norms (N,)
        - frobenius_norm: ||attn1 - attn2||_F
        - correlation: Pearson correlation between flattened matrices
    """
    assert attn1.shape == attn2.shape, "Attention matrices must have same shape"
    assert attn1.ndim == 2, "Expected 2D attention matrices"
    
    # Element-wise difference
    diff = attn1 - attn2
    
    # Per-row residuals (how much each article's attention pattern differs)
    per_row_residuals = np.linalg.norm(diff, axis=1)
    
    # Total Frobenius norm
    frobenius_norm = np.linalg.norm(diff, ord='fro')
    
    # Correlation between matrices
    flat1 = attn1.flatten()
    flat2 = attn2.flatten()
    correlation = np.corrcoef(flat1, flat2)[0, 1]
    
    return {
        'difference_matrix': diff,
        'per_row_residuals': per_row_residuals,
        'frobenius_norm': frobenius_norm,
        'correlation': correlation
    }

def interpret_residuals(
    residuals: np.ndarray,
    metadata: List[Dict],
    top_k: int = 20
) -> Dict:
    """
    Interpret high-residual articles by linking to metadata.
    
    Parameters
    ----------
    residuals : np.ndarray
        Per-point residuals from alignment, shape (N,)
    metadata : List[Dict]
        Article metadata, length N
    top_k : int
        Number of top divergent articles to return
    
    Returns
    -------
    dict
        Analysis of high-divergence articles:
        - top_articles: list of dicts with metadata + residual
        - by_source: residual statistics grouped by source
    """
    # Get top-k indices
    top_indices = np.argsort(residuals)[-top_k:][::-1]
    
    top_articles = []
    for idx in top_indices:
        if idx < len(metadata):
            meta = metadata[idx]
            top_articles.append({
                'index': int(idx),
                'residual': float(residuals[idx]),
                'source': meta.get('source', 'unknown'),
                'title': meta.get('title', '')[:80],
                'url': meta.get('url', '')
            })
    
    # Group by source
    by_source = defaultdict(list)
    for i, meta in enumerate(metadata):
        if i < len(residuals):
            by_source[meta.get('source', 'unknown')].append(residuals[i])
    
    source_stats = {
        source: {
            'mean_residual': float(np.mean(vals)),
            'median_residual': float(np.median(vals)),
            'count': len(vals)
        }
        for source, vals in by_source.items()
    }
    
    return {
        'top_articles': top_articles,
        'by_source': source_stats
    }
